    利用循环遍历每一个li，在定时器回调函数中第一步先清空每个li的高亮样式，赋值全局i变量，i % li.length取余li的总长度，每次的高亮位置即是i的值.

    全局变量rnum表示转几次，当i小于rnum，再次使用定时器。

    当接近rnum时，使用i >= rNum - 8 && i < rNum + that.prizeid，prizeid表示中奖的奖品id，让定时器时间t等于t += (i - rNum + 8) * 5让转圈变慢

    最后that.prizeid && i >= rNum + that.prizeid跳出定时器结束抽奖

        # 抽奖

    play() {
            if(!this.count) {
                return alert('没有次数');
            }
            if(this.count && this.free_spin) {
                this.startscroll();
                this.prizeid = Math.ceil(Math.random()*8);
                console.log(this.prizeid)
            }
        },
        startscroll() {
            this.free_spin = false;
            this.count = this.count - 1;
            let i = 0;
            let t = 100;
            let that = this;
            let round = 4; //表示标记转动的圈数
            let rNum = round * 8;   //表示标记转动的次数
            let li = [];
            for(let j = 1;j < 9;j++) {
                let $li;
                $li = document.getElementById(`opt${j}`)
                li.push($li);
            }
            setTimeout(setFreq, t);
            function setFreq() {　　　　 //首先标记每次刷新就让li所有的class都为空
                for(let j = 0; j < li.length; j++) {
                    li[j].classList.remove('highlight');
                }
                let ord = i % li.length;//该数字就是标记应该出现的位置
                li[ord].classList.add('highlight');
                i++;
                if(i < rNum - 8){
                    setTimeout(setFreq, t);
                }else if(i >= rNum - 8 && i < rNum + that.prizeid){
                    t += (i - rNum + 8) * 5;
                    setTimeout(setFreq, t);
                }
                if( that.prizeid && i >= rNum + that.prizeid){//当i大于转rNum加随机出来的数字次计时器结束，出现信息提示框提示中奖信息
                    setTimeout(() => {
                        that.free_spin=true;
                        let text;
                        that.wheelAwards.some(item => {
                            if(item.pid == `opt${that.prizeid}`){
                                return text=item.text;
                            }
                        })
                        alert(`恭喜！！奖品是${text}`)
                    }, 300);
                }          
            }
        }
