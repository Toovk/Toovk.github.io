    # 1）使用webpack实现热更新

    下载webpack-dev-server插件，会自动进行项目的打包和构建， 配置运行命令，加上serve则不用每次更新都需要输入dev指令

    {
    "scripts": {
        "dev": "webpack serve",

    # 2）使用webpack直接把运行地址指向index.html

    下载html-webpack-plugin插件

    const HTMLPlugin = require('html-webpack-plugin')

const hTMLPlugin = new HTMLPlugin({
    // 指定要复制哪个页面
    template: './src/index.html',
    // 指定复制出来的文件名和存放路径
    filename: './index.html'
})

module.exports = {
    devServer: {
        open: true, //auto open
        host: '127.0.0.1', // taget
        port: 9999, // port
    },
    mode: 'development',
    entry: path.join(__dirname, './src/main.js'),
    output: {
        path: path.join(__dirname, './dist'),
        filename: './main.js'

    // 插件的数组，将来webpack在运行时，会加载并调用这些插件
    plugins: [hTMLPlugin],
    
}

    # 3）使用webpack loader模块加载css、js高级语法以及图片

    module.exports = {
    module: { // 所有第三方文件的匹配规则
        rules: [ // 文件后缀名的匹配规则
            /* 

                1.webpack默认只能打包处理.js结尾的文件，处理不了其它后缀的文件
                2.由于代码中包含index.css文件，webpack处理不了就会查找webpack.config.js这个配置文件
                看module.rules数组中，是否配置了对应的loader加载器材
                3.webpack把index.css先转交给最后一个css-loader处理，处理完毕后把结果转交给下一个style-loader
                4.当style-loader处理完毕后，转交给webpack，合并到/dist/main.js中，最终生成打包好的js文件

            */
            // use的loader顺序不能乱写
            { test: /\.css$/, use:['style-loader', 'css-loader'] },
            { test: /\.less$/, use:[ 'style-loader', 'css-loader', 'less-loader' ] },
            // 调用的只有一个loader，只传递字符串即可，多个就要指定数组
            { test: /\.jpg|jpeg|png|gif$/, use: 'url-loader?limit=22229' },

            /* 
                只需要转换自己打的代码，排除node_modules包
                js高级语法
            */

            { test: /\.js$/, use: 'babel-loader', exclude: /node_modules/ }
        ]
    },
}

    # 4）配置build命令并且每次构建自动删除dist文件夹

    把相关文件构建到对应文件夹

    # package.json:
    "scripts": {
        "build": "webpack --mode production"
    },

# webpack.config.js
    module.exports = {
        output: {
            path: path.join(__dirname, './dist'),
            filename: 'js/main.js'
        },
        module: {
            rules: [
                { test: /\.jpg|jpeg|png|gif$/, use: 'url-loader?limit=22229&outputPath=images' },
            ]
        }
    }

    # 5）devtool打包后调试工具 & 配置@路径

    module.exports = {
    /* 
        加了之后打包方便调试，错误的代码和js行数一致
        devtool: 'eval-source-map'

        实际发布时，建议设置成这个值。报错只暴露行数，不暴露源码
        devtool: 'nosources-source-map',
        
        生产环境下，暴露行数且直接展示具体报错源码
    */
   
    devtool: 'source-map',

    resolve: {
        alias: {
            '@': path.join(__dirname, './src')
        }
    }
}

    # webpack.config.js

    const path = require('path')
const HTMLPlugin = require('html-webpack-plugin')
// 自动删除dist文件夹
const { CleanWebpackPlugin } = require('clean-webpack-plugin');
const hTMLPlugin = new HTMLPlugin({
    // 指定要复制哪个页面
    template: './src/index.html',
    // 指定复制出来的文件名和存放路径
    filename: './index.html'
})
module.exports = {
    devServer: {
        open: true, //auto open
        host: '127.0.0.1', // taget
        port: 9999, // port
    },
    mode: 'development',
    entry: path.join(__dirname, './src/main.js'),
    output: {
        path: path.join(__dirname, './dist'),
        filename: 'js/main.js'
    },
    // html-webpack-plugin可以通过此插件定制index.html的内容
    // 插件的数组，将来webpack在运行时，会加载并调用这些插件
    plugins: [hTMLPlugin, new CleanWebpackPlugin()],
    module: { // 所有第三方文件的匹配规则
        rules: [ // 文件后缀名的匹配规则
            /* 

                1.webpack默认只能打包处理.js结尾的文件，处理不了其它后缀的文件
                2.由于代码中包含index.css文件，webpack处理不了就会查找webpack.config.js这个配置文件
                看module.rules数组中，是否配置了对应的loader加载器材
                3.webpack把index.css先转交给最后一个css-loader处理，处理完毕后把结果转交给下一个style-loader
                4.当style-loader处理完毕后，转交给webpack，合并到/dist/main.js中，最终生成打包好的js文件

            */
            // use的loader顺序不能乱写
            { test: /\.css$/, use:['style-loader', 'css-loader'] },
            { test: /\.less$/, use:[ 'style-loader', 'css-loader', 'less-loader' ] },
            // 调用的只有一个loader，只传递字符串即可，多个就要指定数组
            { test: /\.jpg|jpeg|png|gif$/, use: 'url-loader?limit=22229&outputPath=images' },

            /* 
                只需要转换自己打的代码，排除node_modules包
                js高级语法
            */

            { test: /\.js$/, use: 'babel-loader', exclude: /node_modules/ }
        ]
    },

    /* 
        加了之后打包方便调试，错误的代码和js行数一致
        devtool: 'eval-source-map'

        实际发布时，建议设置成这个值。报错只暴露行数，不暴露源码
        devtool: 'nosources-source-map',
        
        生产环境下，暴露行数且直接展示具体报错源码
    */
   
    devtool: 'source-map',
    
    resolve: {
        alias: {
            '@': path.join(__dirname, './src')
        }
    }
}
// webpack-dev-server会自动进行项目的打包和构建
